You can also use for loops to iterate data frames.

# create a data frame
c1<-c("AA","BB","CC")
c2<-c(11,22,33)
df<-data.frame(c1,c2)

# display data frame
df

# len() returns the number of columns
len(df)
# nrow() returns the number of rows
nrow(df)

traverseDF <- function() {
  n<-nrow(df)
  for (i in 1:n) {
    # access column 2 in row i
    print(df[i,2])
    # or this way
    print(df$c2[i])
  }
}
